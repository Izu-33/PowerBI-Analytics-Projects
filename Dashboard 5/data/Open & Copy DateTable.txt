Calendar = 
ADDCOLUMNS(
 CALENDARAUTO(),
    "Year", YEAR([Date]),
    "Month", FORMAT([Date],"mmm"),
    "Month num", MONTH([Date]),
    "quarter", FORMAT([Date], "\QQ"),
    "weekdays", FORMAT([Date],"ddd"),
    "Week num", WEEKDAY([Date])
)
